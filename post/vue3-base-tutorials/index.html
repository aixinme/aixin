基本框架

    <p>{{ message }}</p>
    <p>{{ content() }}</p>
</div>
var vm = new Vue({
    el: '#app',
    data: {
        message: 'Hello'
    },
    methods: {
        content: function(){
            return this.message + ' Vue.js';
        }
    }
})
说明：

vue代码结构：

new Vue({
    el: '#app',//根元素
    data: { //变量
        message: 'Hello'
    },
    methods: { //函数方法
        content: function(){
            return this.message + ' Vue.js';//this可以获取当前vue中所有的变量或函数
        }
    },
    filters:{ //过滤器
        filterA: function(arg1,arg2){
            return arg1 + arg2;
        },
        filterB: function(){
            //...
        }
    }
})

//可以将上面代码传给一个对象，比如
var vm = new Vue({...

//这样就可以对象获取或操作里面的属性，比如
vm.message = 'Message has been changed!';

//可以用$获取对象中的data
vm.$data
//或者
vm.$el

// 双大括号显示内容
<p>{{ message }}</p> // 变量直接写变量名
<p>{{ content() }}</p> // 函数直接写函数名()，记得后面跟小括号

模板语法

==v-html 指令==

<div v-html="message"></div>
// 通过v-html来输出html内容
// {{ message}} 只支持文本

==v-bind 指令==

//绑定属性要用冒号:
<div v-bind:id="'list-' + id">菜鸟教程</div>
// id从data中获取
<a v-bind:href="url">菜鸟教程</a>
//可以绑定所有元素属性，比如src、href、class、id等等

//v-bind的缩写形式，直接用冒号代替
<a :href="url"></a>

==运行JavaScript表达式==

{{5+5}}<br>
{{ ok ? 'YES' : 'NO' }}<br>
{{ message.split('').reverse().join('') }}

==v-if 条件判断==

<p v-if="seen">现在你看到我了</p>

//可以配合v-else使用
<div v-else>
    Nothing
</div>

//也可以用v-else-if，同v-if

// 相关方法 v-show
<h1 v-show="ok">Hello!</h1>

==v-on 监听DOM事件==

<a v-on:click="doSomething">

//修饰符
<form v-on:submit.prevent="onSubmit"></form>

//v-on的缩写形式，直接用@符号代替
<a @click="doSomething"></a>

==v-model 双向数据绑定==

<input v-model="message">
// v-model 指令用来在 input、select、textarea、checkbox、radio 等表单控件元素上创建双向数据绑定，
// 根据表单上的值，自动更新绑定的元素的值。

==v-for条件循环==

// 循环数组
<li v-for="site in sites">
   {{ site.name }}
</li>
data: {
    sites: [
      { name: 'Runoob' },
      { name: 'Google' },
      { name: 'Taobao' }
    ]
}

// 循环对象
<li v-for="value in object">
    {{ value }}
</li>

data: {
    object: {
      name: '菜鸟教程',
      url: 'http://www.runoob.com',
      slogan: '学的不仅是技术，更是梦想！'
    }
}

// 值，键，索引
<li v-for="(value, key, index) in object">
    {{ index }}. {{ key }} : {{ value }}
</li>

// 循环整数
<li v-for="n in 10">
    {{ n }}
</li>

过滤器

{{ message | filterA}}
{{ message | filterA | filterB }}
{{ message | filterA('arg1', arg2) }}

// 过滤器就是JavaScript的函数，同时也可以给过滤器传参arg1、arg2等
