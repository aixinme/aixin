@Service
public class ArticleServiceImpl implements ArticleService {
    @Autowired
    private ArticleMapper articleMapper;
    @Override
    public List<ArticleDTO> getArticleList() {
        // 查询所有文章
        List<Article> articles = articleMapper.selectList(null);
        // 定义顶级文章列表存储对象
        List<ArticleDTO> articleDTOs = new ArrayList<>();
        // 定义一个map，用于存储文章的父子关系
        Map<Long, ArticleDTO> articleDTOMap = new HashMap<>();
        // 遍历文章，将文章按照父子关系存储到map中
        for (Article article : articles) {
            // 定义一个文章DTO对象
            ArticleDTO articleDTO = new ArticleDTO();
            // 将文章属性复制到文章DTO对象中
            BeanUtils.copyProperties(article, articleDTO);
            // 再将文章DTO对象存储到map中
            articleDTOMap.put(article.getId(), articleDTO);

            // 判断文章是否为顶级文章
            if (article.getParentId() == null) {
                // 如果是顶级文章，直接将文章DTO对象存储到顶级文章列表中
                articleDTOs.add(articleDTO);
            } else {
                // 如果不是顶级文章，将文章DTO对象存储到父文章的子文章列表中
                ArticleDTO parentArticleDTO = articleDTOMap.get(article.getParentId());
                // 判断父文章是否存在
                if (parentArticleDTO != null) {
                    // 如果父文章存在，将文章DTO对象存储到父文章的子文章列表中
                    parentArticleDTO.getChildren().add(articleDTO);
                }
        }
    }
    // 返回顶级文章列表
    /**
     * 输出结果：
     * [
     *     {
     *         "id": 1,
     *         "title": "文章1",
     *         "content": "文章1的内容",
     *         "children": [
     *             {
     *                 "id": 2,
     *                 "title": "文章2",
     *                 "content": "文章2的内容",
     *                 "children": []
     *             }
     *         ]
     *     }
     * ]
     */
    return articleDTOs;
}

BeanUtils.copyProperties(article, articleDTO);将文章属性复制到文章DTO对象中对应属性中

  • BeanUtils.copyProperties 不会复制静态字段、final 字段以及没有 getter/setter 方法的字段。
  • 如果需要排除某些属性不进行复制，可以使用 BeanUtils.copyProperties 的第三个参数指定要忽略的属性名数组。

当然也可以使用geter/setter方法进行复制，比如：articleDTO.setName(article.getName());逐个复制每个属性
