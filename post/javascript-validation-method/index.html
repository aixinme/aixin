// 是否为空
vernull = function(value){
　　if(value.trim(value).length == 0){
　　　　return false;
　　}else{
　　　　return true;
　　}
}
// 身份证校验
veridc = function(value){
　　var ext = /(^\d{15}&)|(^\d{18})|(^\d{17}(\d|X|x)$)/;
　　return ext.test(value);
}

 
// 英文校验
veren = function(value){
　　var ext = /(^[a-zA-Z]+$)/;
　　return ext.test(value);
}

 
// 数字校验
vernum = function(value){
　　var ext = /^[0-9]*$/;
　　return ext.test(value);
}

 // 小数校验
verdou = function(value){
　　var ext = /^[0-9]+(.[0-9]{1,3})?$/;
　　return ext.test(value);
}

 
// 邮箱校验
vermail = function(value){
　　var ext = /^([a-zA-Z0-9_-])+@([a-zA-Z0-9_-])+((\.[a-zA-Z0-9_-]{2,3}){1,2})$/;
　　return ext.test(value);
}

 
// 手机号校验
vermob =function(value){
　　var ext = /^1[3456789][0-9]{9}$/;
　　return ext.test(value);
}
